# 一、安装mongodb

参考文档https://www.mongodb.com/docs/manual/tutorial/install-mongodb-on-red-hat-tarball/ (opens new window) 配置：https://www.mongodb.com/docs/v5.0/reference/configuration-file-settings-command-line-options-mapping/ (opens new window) 主节点：172.18.12.224:27017 从节点：172.18.12.225:27017 仲裁节点：172.18.120.201:27017

# 1. 安装MongoDB .tgz所需依赖

sudo yum install libcurl openssl xz-libs -y
1

下载地址： ➤ MongoDB安装包下载中心 (opens new window) ➤Mongosh下载中心 (opens new window)

cd /data/mongodb
tar -zxvf mongodb-linux-*-5.0.9.tgz
mv mongodb-linux-x86_64-rhel70-5.0.9/* ./
ln -s $(pwd)/bin/* /usr/local/bin/

tar -zxvf mongosh-1.5.0-linux-x64.tgz
cd mongosh-1.5.0-linux-x64
ln -s $(pwd)/bin/* /usr/local/bin/

cd  /data/mongodb
mkdir data
mkdir logs

groupadd mongod
useradd -g mongod mongod -s /sbin/nologin
chown -R mongod.mongod /data/mongodb

[Service]  
Type=forking

PrivateTmp=true 

[Install]  
WantedBy=multi-user.target

# 2. 第二步：配置开启副本集验证

# more mongod.conf 
systemLog:
  destination: file
  path: "/data/mongodb/logs/mongod.log"
  logAppend: true
storage:
  dbPath: "/data/mongodb/data"
  journal:
    enabled: true
net:
  bindIp: localhost,172.18.12.224
  port: 27017
processManagement:
  fork: true
security:
#开启副本集验证
  keyFile: /data/mongodb/mongo.keyfile
#开启账号验证
  authorization: enabled

replication:
  replSetName: blm

# 四、SpringDataMongoDB连接副本集

#使用用户名和密码连接到 MongoDB 服务器，你必须使用
#'username:password@hostname/dbname' 格式，'username'为用户名，'password' 为密码。
#目标：使用用户bobo使用密码 123456 连接到MongoDB 服务上。
application.yml：
  spring:
  #数据源配置
    data:
      mongodb:
      #副本集有认证的情况下，字符串连接
        uri:mongodb://root:123456@172.18.12.224:27017,172.18.12.225:27017,172.18.120.201:27017/集合名?connect=replicaSet&slaveOk=true&replicaSet=blm
