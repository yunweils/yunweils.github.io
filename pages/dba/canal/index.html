# Mysql+Canal+Kafka+Zookeeper

# 一、zookeeper

说明：版本选择查看kafka中自带zookeeper版本

# 1.1安装jdk

下载jdk安装包jdk-8u202-linux-x64.tar.gz

#解压安装包
tar -zxvf jdk-8u202-linux-x64.tar.gz

#配置环境变量
vim /etc/profile
export JAVA_HOME=/data/pre_dw/jdk1.8.0_202
export JRE_HOME=${JAVA_HOME}/jre
export CLASSPATH=.:${CLASSPATH}:${JAVA_HOME}/lib:${JRE_HOME}/lib
export PATH=${PATH}:${JAVA_HOME}/bin:${JRE_HOME}/bin

#环境变量生效
source /etc/profile

# 1.2安装zookeeper

下载地址：https://www.apache.org/dyn/closer.lua/zookeeper/zookeeper-3.5.9/apache-zookeeper-3.5.9-bin.tar.gz (opens new window) 下载安装包apache-zookeeper-3.5.9-bin.tar.gz

#解压zookeeper安装包
tar -zxvf apache-zookeeper-3.5.9-bin.tar.gz

#修改配置文件
cd apache-zookeeper-3.5.9-bin/conf

vim zoo.cfg
tickTime=2000
initLimit=10
syncLimit=5
#数据保存位置
dataDir=/data/zookeeper/zookeeper_data
dataLogDir=/data/zookeeper/zookeeper_logs
clientPort=2181

#启动zookeeper
/data/zookeeper/apache-zookeeper-3.5.9-bin/bin/zkServer.sh  start

# 二、kafka

#解压安装包
tar -zxvf kafka_2.13-2.8.0.tgz

#配置修改
cd kafka_2.13-2.8.0/config

vim server.properties
broker.id=1
listeners=PLAINTEXT://0.0.0.0:9092
advertised.listeners=PLAINTEXT://192.168.10.131:9092
num.network.threads=3
num.io.threads=8
socket.send.buffer.bytes=102400
socket.receive.buffer.bytes=102400
socket.request.max.bytes=104857600
log.dirs=/data/kafka/kafka_2.13-2.8.0/kafka-logs
num.partitions=1
num.recovery.threads.per.data.dir=1
offsets.topic.replication.factor=1
transaction.state.log.replication.factor=1
transaction.state.log.min.isr=1
log.retention.hours=168
log.segment.bytes=1073741824
log.retention.check.interval.ms=300000
zookeeper.connect=192.168.10.131:2181
zookeeper.connection.timeout.ms=18000
group.initial.rebalance.delay.ms=0
##消息长度修改
message.max.bytes=104858800


#启动kafka
/data/kafka/kafka_2.13-2.8.0/bin/kafka-server-start.sh -daemon /data/kafka/kafka_2.13-2.8.0/config/server.properties

# 三、canal

下载安装包canal.deployer-1.1.5.tar.gz

#配置kafka链接
cd /data/canal/conf/
vim canal.properties
# tcp, kafka, RocketMQ 这里选择kafka模式
canal.serverMode = kafka
##################################################
#########                    Kafka                   #############
##################################################
kafka.bootstrap.servers = 192.168.10.131:9092
kafka.acks = all
kafka.compression.type = none
kafka.batch.size = 16384
kafka.linger.ms = 1
kafka.max.request.size = 10485760
kafka.buffer.memory = 33554432
kafka.max.in.flight.requests.per.connection = 1
kafka.retries = 0
kafka.kerberos.enable = false
kafka.kerberos.krb5.file = "../conf/kerberos/krb5.conf"
kafka.kerberos.jaas.file = "../conf/kerberos/jaas.conf"

# 方法一 canal采集多库并写入单库：

配置链接主库 写入kafka设置

##解压安装包
cd /data/canal
tar -zxvf canal.deployer-1.1.5.tar.gz

#修改配置文件
cd conf/example

vim instance.properties
canal.instance.gtidon=false
#主库IP+PORT
canal.instance.master.address=192.168.10.131:3306
#主库的二进制文件以及position
canal.instance.master.journal.name=mybinlog.000749
