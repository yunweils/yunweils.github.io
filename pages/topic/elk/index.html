Filebeat+Kafka+ELK 7.17.6部署

#Filebeat+Kafka+ELK 7.17.6部署

本篇教程在示例步骤中使用了以下版本的软件。操作时，请您以实际软件版本为准。

  • 操作系统：CentOS 7.9 64位
  • filebeat：7.17.6
  • zookeeper：3.6.3
  • kafka：2.13-3.2.3
  • logstash：7.17.6
  • elasticsearch：7.17.6
  • kibana：7.17.6

注意：filebeat、logstash、elasticsearch、kibana 版本统一

机器： 172.16.16.11：filebeat 172.16.16.6：zookeeper、kafka、logstash、elasticsearch、kibana

# 三、Zookeeper+Kafka安装

# 1. zookeeper安装

cd /data/elk/apache-zookeeper-3.6.3-bin

#创建数据和日志目录
mkdir -p data  logs

#创建配置文件
cp config/zoo_sample.cfg  config/zoo.cfg

#修改配置文件
vim config/zoo.cfg
# The number of milliseconds of each tick
tickTime=2000
# The number of ticks that the initial 
# synchronization phase can take
initLimit=10
# The number of ticks that can pass between 
# sending a request and getting an acknowledgement
syncLimit=5
# the directory where the snapshot is stored.
# do not use /tmp for storage, /tmp here is just 
# example sakes.
dataDir=/data/elk/apache-zookeeper-3.6.3-bin/data
dataLogDir=/data/elk/apache-zookeeper-3.6.3-bin/logs
# the port at which the clients will connect
clientPort=2181

#启动
/data/elk/apache-zookeeper-3.6.3-bin/bin/zkServer.sh  start

# 2.kafka安装

cd /data/elk/kafka_2.13-3.2.3/

#创建数据日志目录
mkdir -p kafka-logs

#新增hosts
vim /etc/hosts
172.16.16.6 elk-kafka1

#修改配置文件
vim config/server.properties
broker.id=1
listeners=PLAINTEXT://0.0.0.0:9092
advertised.listeners=PLAINTEXT://elk-kafka1:9092
num.network.threads=3
num.io.threads=8
socket.send.buffer.bytes=102400
socket.receive.buffer.bytes=102400
socket.request.max.bytes=104857600
log.dirs=/data/elk/kafka_2.13-3.2.3/kafka-logs
num.partitions=1
num.recovery.threads.per.data.dir=1
offsets.topic.replication.factor=1
transaction.state.log.replication.factor=1
transaction.state.log.min.isr=1
log.retention.hours=168
log.segment.bytes=1073741824
log.retention.check.interval.ms=300000
zookeeper.connect=localhost:2181
zookeeper.connection.timeout.ms=18000
group.initial.rebalance.delay.ms=0
##消息长度
message.max.bytes=104858800

#启动kafka
/data/elk/kafka_2.13-3.2.3/bin/kafka-server-start.sh -daemon /data/elk/kafka_2.13-3.2.3/config/server.properties

# 四、Filebeat安装

#添加hosts
vim /etc/hosts
172.16.16.6 elk-kafka1

#解压tar.gz
cd /data
tar -zxvf filebeat-7.17.6-linux-x86_64.tar.gz
cd filebeat-7.17.6-linux-x86_64

#修改配置
vim filebeat.yml
##参考https://www.elastic.co/guide/en/beats/filebeat/current/filebeat-input-filestream.html
filebeat.inputs:
##官方推荐filestream替代log
- type: filestream
##id必须有，省略或更改id可能会导致采集数据重复
  id: blm_smsaas
  enabled: true
##数据采集路径
  paths:
    - /data/logs/blm_smsaas/*.log
  prospector.scanner.exclude_files: ['.*-error.*.log']
##额外信息输出
  fields:
    log_topic: blm_smsaas
    ip: 112.74.100.208
##标签
  tags: ["blm_smsaaas","blm"]
##ignore_older必须大于close_inactive
##ignore_older表示只抓取最近24h内更新文件
  ignore_older: '25h'
  close.on_state_change.inactive: '24h'
##对报错多行日志进行处理，以时间为开头来分割
##参考https://www.elastic.co/guide/en/beats/filebeat/current/multiline-examples.html
  parsers:
  - multiline:
      type: pattern
      pattern: '^[0-9]{4}-[0-9]{2}-[0-9]{2}'
      negate: true
      match: after
##排除DEBUG日志，正则匹配
  exclude_lines: ['DEBUG']

- type: filestream
  id: blm_smmobile
  enabled: true
  paths:
    - /data/logs/blm_smsaas/*.log
  prospector.scanner.exclude_files: ['.*-error.*.log']
  fields:
    log_topic: blm_smmobile
  tags: ["blm_smmobile"]
  ignore_older: '25h'
  close.on_state_change.inactive: '24h'
  parsers:
  - multiline:
      type: pattern
      pattern: '^[0-9]{4}-[0-9]{2}-[0-9]{2}'
      negate: true
      match: after
  exclude_lines: ['DEBUG']

##参考：https://www.elastic.co/guide/en/beats/filebeat/current/kafka-output.html
output.kafka:
##参考：https://www.elastic.co/guide/en/beats/filebeat/current/configuration-output-codec.html
  codec.json:
    pretty: true
    escape_html: false
##kafka地址
  hosts: ["elk-kafka1:9092"]
##kafka的主题topic
  topic: '%{[fields.log_topic]}'
##最大消息长度
  max_message_bytes: 10000000

#启动
nohup /data/filebeat-7.17.6-linux-x86_64/filebeat -c /data/filebeat-7.17.6-linux-x86_64/filebeat.yml >  /data/filebeat-7.17.6-linux-x86_64/nohup.out &

#验证，上kafka查看是否接收到消息
##列出已创建的topic列表
./bin/kafka-topics.sh --list --bootstrap-server elk-kafka1:9092
##模拟客户端去接受消息
./bin/kafka-console-consumer.sh --bootstrap-server elk-kafka1:9092 --topic blm_smsaas --from-beginning

# 七、Logstash安装

cd /data/elk/logstash-7.17.6
#修改配置文件
vim config/logstash-sample.conf
input {
  kafka {
    bootstrap_servers => "elk-kafka1:9092"

    group_id => "blm_logstash"
    #topics => ["blm_smsaas"]
    topics_pattern => ".*"
    #自动将偏移量重置为最早的偏移量
    ##参考https://www.elastic.co/guide/en/logstash/current/plugins-inputs-kafka.html#plugins-inputs-kafka-auto_offset_reset
    auto_offset_reset => "earliest"

    #json格式
    codec => json
    decorate_events => true
  }

}

##message切割和提取，用来定义kibana中的展示字段
filter {
  grok{
    match => {"message" => "%{TIMESTAMP_ISO8601:time} \[%{DATA:thread}\] %{LOGLEVEL:level} %{GREEDYDATA:data}"}
  }

  ##@timestamp同步为日志时间
  date {
    match => [ "time","ISO8601" ]
    target => "@timestamp"
  }

}




output {
  elasticsearch {
    ##elasticsearch地址
    hosts => ["http://localhost:9200"]
    ##%{[@metadata][kafka][topic]}必须开启decorate_events才生效，动态获取kafka的topic
    ##elasticsearch-curator按索引名字时间操作索引，所以加上%{+YYYY.MM.dd}
    index => "%{[@metadata][kafka][topic]}-%{+YYYY.MM.dd}"

  }
}

#启动
/data/elk/logstash-7.17.6/bin/logstash -f /data/elk/logstash-7.17.6/config/logstash-sample.conf &

# 九、elasticsearch-curator

注意 elasticsearch-curator的版本与elasticsearch版本兼容 https://www.elastic.co/guide/en/elasticsearch/client/curator/5.6/version-compatibility.html#version-compatibility (opens new window) elasticsearch-curator直接操作索引，索引最好加时间，通过时间处理索引 elasticsearch中索引删除

#安装curator
#https://www.elastic.co/guide/en/elasticsearch/client/curator/5.6/python-source.html
pip install elasticsearch-curator


vim config.yml
client:
  hosts:
    - 127.0.0.1
  port: 9200      
  url_prefix:
  use_ssl: False
  certificate:
  client_cert:
  client_key:
  ssl_no_validate: False
  http_auth: elastic:Dev#21
  timeout: 30
  master_only: False
logging:
  loglevel: INFO
  logfile: /data/elk/es-curator/curator.log
  logformat: default
  blacklist: ['elasticsearch', 'urllib3']


vim action_file.yml
---
actions:
  1:
    action: delete_indices
    description: "Clean up ES by deleting old indices"
    options:
      timeout_override:
      continue_if_exception: False
      disable_action: False
      ignore_empty_list: True
    filters:
    - filtertype: age
      source: name
      direction: older
      timestring: '%Y.%m.%d'
      unit: days
      unit_count: 15
      field:
      stats_result:
      epoch:
      exclude: False


#添加定时任务，删除15天前索引
curator --config /data/elk/es-curator/config.yml  /data/elk/es-curator/action_file.yml
